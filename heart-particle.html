Heart Particle Animation
/* PARTICLES */
const particles = [];
const heartPath = getHeartPath();
for (let i = 0; i < heartPath.length; i++) {
  const p = heartPath[i];
  particles.push({
    x: p.x + (Math.random() - 0.5) * 30,
    y: p.y + (Math.random() - 0.5) * 30,
    targetX: p.x, targetY: p.y,
    delay: i * 0.002,
    duration: Math.random() * 3 + 2
  });
}
